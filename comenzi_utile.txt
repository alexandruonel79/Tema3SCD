SELECT 
  last("data") AS "Current", 
  min("data") AS "Min", 
  max("data") AS "Max", 
  mean("data") AS "Avg"
FROM /.*/ 
WHERE "metric" = 'BAT' AND $timeFilter
GROUP BY "device"
